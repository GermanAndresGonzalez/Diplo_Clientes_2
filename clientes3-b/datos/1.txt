def obtener_id_de_fila(self, tree, i):
    # Obtener la lista de IDs de los ítems en el Treeview
    lista_ids = tree.get_children()
    # Verificar si el índice i está dentro del rango de la lista de IDs
    if i < len(lista_ids):
        # Obtener el ID de la fila correspondiente
        id_fila = lista_ids[i]
        return id_fila
    else:
        return None
